The Event Message Header Schema

Type: object

Type: string Must match regular expression: [0-9]{4}-[0-9]{2}-[0-9]{2}T[0-9]{2}:[0-9]{2}:[0-9]{2}[.]?[0-9]{0,}?Z
Example:

"2019-12-26T12:47:23.1234Z"

Type: enum (of string)

Must be one of:

  • "control"
  • "observation"
  • "chat"
  • "status"
Must match regular expression: ^([a-z_]*?)$
Example:

"observation"

Type: string Must match regular expression: ^([0-9]*?)\.([0-9]*?)$
Examples:

"0.1"
"1.0"

Type: object

Type: string Default: "" Must match regular expression: [0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}
Example:

"123e4567-e89b-12d3-a456-426655440000"

Type: string Default: "" Must match regular expression: [0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}
Example:

"123e4567-e89b-12d3-a456-426655440000"

Type: string Default: "" Must match regular expression: [0-9]{4}-[0-9]{2}-[0-9]{2}T[0-9]{2}:[0-9]{2}:[0-9]{2}[.]?[0-9]{0,}?Z
Example:

"2019-12-26T14:05:02.3412Z"

Type: string Default: "" Must match regular expression: ^(.*)$
Example:

"simulator"

Type: string Default: "" Must match regular expression: ^([a-zA-Z0-9_]*?)$
Examples:

"equip"
"triage"

Type: string Default: "" Must match regular expression: ^([0-9]*?)\.([0-9]*?)$
Examples:

"0.1"
"1.0"
